type(course)

len(course)

course_r <- "R programming"

typeof(course_r)

length(course_r)

nchar(course_r)
## [1] 13

1.0.1 Slicing

course[0]
## 'P'
course[-1]
## 'g'
course[0:3]
## 'Pyt'
course[:3]
## 'Pyt'
course[0:]
## 'Python programming'
course[:]
## 'Python programming'
substr(course_r, 1, 1)
## [1] "R"
substr(course_r, 1, 4)
## [1] "R pr"

1.0.3 Formatted String

first = "Kittipos"
last = "Sirivongrungson"

print(f"{first} {last} is {len(first + last)} letter long")
## Kittipos Sirivongrungson is 23 letter long
first <-  "Kittipos"
last <-  "Sirivongrungson"

glue::glue("{first} {last} is {nchar(paste0(first, last))} letter long")
## Kittipos Sirivongrungson is 23 letter long
